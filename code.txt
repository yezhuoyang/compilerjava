

int N = 8;
int[] row = new int[8];
int[] col = new int[8];

void printBoard() {
  int i;
  int j;
  for (i = 0; i < N; i++) {
    for (j = 0; j < N; j++) {
      if (col[i] == j)
        print(" O");
      else
        print(" .");
    }
    println("");
  }
  println("");
}


int main(){
  int i;
  for (i = 0; i < 8; ++i) {
    row[i] = 0;
    col[i] = 0;
  }

  printBoard();
  printBoard();



}
